Setting up recipes and workflows for the three models

# Recipe for linear regression and LASSO
lm_lasso_recipe <- recipe(Y ~ ., data = analysis_df) |>
  step_dummy(all_nominal_predictors()) |>
  step_zv(all_predictors()) |>
  step_lincomb(all_predictors()) |>
  step_normalize(all_numeric_predictors())

# Recipe for random forest
rf_recipe <- recipe(Y ~ ., data = analysis_df) |>
  step_zv(all_predictors())

# Model 1: ordinary linear regression

  set_engine("lm")

Make predictions on the full dataset and compute RMSE

# Helper function to add predictions and compute RMSE
get_apparent_rmse <- function(fitted_wf, data, model_name) {
  preds <- predict(fitted_wf, new_data = data) |>
    bind_cols(data |> select(Y))

  rmse_val <- rmse(preds, truth = Y, estimate = .pred)
  rmse_val$model <- model_name

  list(preds = preds, rmse = rmse_val)
}

lm_results <- get_apparent_rmse(lm_fit, analysis_df, "Linear model")
lasso_results <- get_apparent_rmse(lasso_fit, analysis_df, "LASSO (penalty = 0.1)")
rf_results <- get_apparent_rmse(rf_fit, analysis_df, "Random forest")

rmse_table <- bind_rows(

Observed versus predicted plots

plot_obs_pred <- function(pred_df, model_title) {
  ggplot(pred_df, aes(x = Y, y = .pred)) +
    geom_point(alpha = 0.6) +
    geom_abline(intercept = 0, slope = 1, linetype = 2) +
    labs(
      title = paste("Observed vs Predicted:", model_title),
      x = "Observed Y",
      y = "Predicted Y"
    ) +
    theme_minimal()
}

plot_obs_pred(lm_results$preds, "Linear model")

# Create a tunable LASSO model
lasso_tune_spec <- linear_reg(
  penalty = tune(),
  mixture = 1
) |>
  set_engine("glmnet")

lasso_tune_wf <- workflow() |>
  add_recipe(lm_lasso_recipe) |>
  add_model(lasso_tune_spec)

# Grid of 50 penalty values on the log10 scale
lasso_grid <- tibble(
  penalty = 10^seq(log10(1e-5), log10(1e2), length.out = 50)
)

# Apparent resample object
apparent_resamples <- apparent(analysis_df)

# Tune over the grid
lasso_apparent_tuned <- tune_grid(
  lasso_tune_wf,
  resamples = apparent_resamples,
  grid = lasso_grid,
  metrics = metric_set(rmse),
  control = control_grid(save_pred = TRUE)
)

# For apparent() there is only one resample, so keep the unsummarized results
